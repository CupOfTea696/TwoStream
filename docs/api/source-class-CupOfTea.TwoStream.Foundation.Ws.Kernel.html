 78         
 79         try {
 80             $response = $this->sendRequestThroughRouter($request);
 81             $response = $response->getContent();
 82             
 83             $json = json_decode($response, true);
 84             if (json_last_error() == JSON_ERROR_NONE) {
 85                 $response = $json;
 86             }
 87         } catch (Exception $e) {
 88             if ($e instanceof NotFoundHttpException) {
 89                 $response = $e;
 90             } else {
 91                 throw $e;
 92             }
 93         }
 94         
 95         $this->app['events']->fire('wskernel.handled', [$request, $response]);

 97         return $response;
